\item Microsoft Visual Studio with \href{https://www.visualmicro.com/}{\texttt{ Visual Micro}}
\end{DoxyItemize}

\begin{DoxyItemize}
\item A\+Tmega328P
\item A\+Tmega2560
\item A\+RM Cortex M0
\item E\+S\+P8266
\item S\+T\+M32
\end{DoxyItemize}

{\bfseries{Web\+Socket\+Server compatible browsers\+:}}
\begin{DoxyItemize}
\item Chrome
\item Edge
\item Firefox
\end{DoxyItemize}
